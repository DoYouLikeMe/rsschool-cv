#### Maximum Triplet Sum

```
function maxTriSum(numbers){
  const unique = numbers.filter((item, i, ar) => ar.indexOf(item) === i);
  const sortedUniqueTriplet = unique.sort((a, b) => b - a).splice(0, 3);
  const tripletSum = sortedUniqueTriplet.reduce(
    (previousValue, currentValue) => previousValue + currentValue,
    0
  );

  return tripletSum;

}

```
